//이미지 슬라이드 하는 부분 다시 확인하고 작성

const popupSlide = document.querySelector('.popup-slide');
const slides = document.querySelector('.slides');
const slideItem = document.querySelectorAll('.slide-item');

let currentIndex = 0;

slideItem.forEach(inner => {
    slideItem.style.width = `${popupSlide.clientWidth}px`
});

slides.style.width = `${popupSlide.clientWidth * slideItem.length}px`;

const prev = document.querySelector('.prev');
const next = document.querySelector('.next');

prev.addEventListener('click', ()=>{
    currentIndex--;
    currentIndex = currentIndex < 0 ? 0 : currentIndex;
    slides.style.marginLeft = `-${popupSlide.clientWidth * currentIndex}px`;
});

next.addEventListener('click', ()=>{
    currentIndex++;
    currentIndex =
        currentIndex >= slideItem.length ? slideItem.length -1 : currentIndex;
        slides.style.marginLeft = `-${popupSlide.clientWidth * currentIndex}px`;
});
